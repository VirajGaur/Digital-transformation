CREATE TABLE `Customer` (
	`CustomerID` INT(11) UNSIGNED NOT NULL AUTO_INCREMENT,
	`FirstName` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`LastName` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`EmailAddress` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`Password` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	PRIMARY KEY (`CustomerID`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=9
;

CREATE TABLE `GroceryList` (
	`ListID` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
	`ProductID` INT(10) UNSIGNED NULL DEFAULT NULL,
	`Price` FLOAT UNSIGNED NULL DEFAULT NULL,
	`Quantity` INT(10) UNSIGNED NULL DEFAULT NULL,
	`Total` FLOAT UNSIGNED NULL DEFAULT NULL,
	`CustomerID` INT(11) UNSIGNED NOT NULL,
	PRIMARY KEY (`ListID`) USING BTREE,
	INDEX `Product key 1` (`ProductID`) USING BTREE,
	INDEX `Customer key 1` (`CustomerID`) USING BTREE,
	CONSTRAINT `Customer key 1` FOREIGN KEY (`CustomerID`) REFERENCES `CSC4008`.`Customer` (`CustomerID`) ON UPDATE NO ACTION ON DELETE NO ACTION,
	CONSTRAINT `Product key 1` FOREIGN KEY (`ProductID`) REFERENCES `CSC4008`.`Product` (`ProductID`) ON UPDATE CASCADE ON DELETE CASCADE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=9
;

CREATE TABLE `Inventory` (
	`StoreID` INT(10) UNSIGNED NOT NULL,
	`ProductID` INT(10) UNSIGNED NOT NULL,
	`Total_Quantity` INT(10) UNSIGNED NULL DEFAULT NULL,
	`Availability` BINARY(50) NULL DEFAULT NULL,
	`Co-ordinate X` INT(10) UNSIGNED NULL DEFAULT NULL,
	`Co-ordinate Y` INT(10) UNSIGNED NULL DEFAULT NULL,
	PRIMARY KEY (`StoreID`, `ProductID`) USING BTREE,
	UNIQUE INDEX `StoreID_ProductID` (`StoreID`, `ProductID`) USING BTREE,
	INDEX `Product key` (`ProductID`) USING BTREE,
	CONSTRAINT `Product key` FOREIGN KEY (`ProductID`) REFERENCES `CSC4008`.`Product` (`ProductID`) ON UPDATE CASCADE ON DELETE CASCADE,
	CONSTRAINT `Store key` FOREIGN KEY (`StoreID`) REFERENCES `CSC4008`.`Store` (`StoreID`) ON UPDATE CASCADE ON DELETE CASCADE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
;

CREATE TABLE `Product` (
	`ProductID` INT(11) UNSIGNED NOT NULL,
	`ProductName` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`Weight` DECIMAL(10,3) UNSIGNED NULL DEFAULT NULL,
	`Price` DECIMAL(10,2) UNSIGNED NULL DEFAULT NULL,
	`QuantityPerProduct` INT(11) UNSIGNED NULL DEFAULT NULL,
	`Image` TEXT NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	PRIMARY KEY (`ProductID`) USING BTREE,
	UNIQUE INDEX `idxProductID` (`ProductID`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
;

CREATE TABLE `Store` (
	`StoreID` INT(11) UNSIGNED NOT NULL,
	`StoreName` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`AddressLine1` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`AddressLine2` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`City` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`Country` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`Postcode` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`Blueprint` BLOB NULL DEFAULT NULL,
	PRIMARY KEY (`StoreID`) USING BTREE,
	UNIQUE INDEX `idxStoreID` (`StoreID`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
;
